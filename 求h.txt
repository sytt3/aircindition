import math
def h(t,fy):
    B=101325
    c8=-5800.2206
    c9=1.3914993
    c10=-0.048640239
    c11=0.41764768*(10**-4)
    c12=-0.14452093*(10**-7)
    c13=6.5459673
    T=273.15+t
    Pqb=math.exp(c8/T+c9+c10*T+c11*(T**2)+c12*(T**3)+c13*math.log(T))
    db=622*Pqb/(B-Pqb)
    d=db*fy
    h=1.005*t+d*(2500+1.84*t)/1000
    return h
print(h(34,0.65))
